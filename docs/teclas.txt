ARRIBA = 273 = flecha arriba
mueve al héroe hacia arriba de la pantalla, a la vez que desplaza el mapa hacia abajo

ABAJO = 274 = flecha abajo
mueve al héroe hacia abajo de la pantalla, a la vez que desplaza el mapa hacia arriba

DERECHA = 275 = flecha izquierda
mueve al héroe hacia la izquierda de la pantalla, a la vez que desplaza el mapa hacia la derecha

IZQUIERDA = 276 = flecha derecha
mueve al héroe hacia la derecha de la pantalla, a la vez que desplaza el mapa hacia la izquierda
        
ACCION = 120 = tecla X
dispara la acción por defecto del héroe. por el momento, no ocurre nada.

MENU = 13 = tecla enter
abre el menú. por el momento, solo devuelve un mensajito a la consola

SALIR = 27 = tecla escape
cierra el juego
